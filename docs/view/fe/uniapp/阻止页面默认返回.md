
##### uniapp阻止页面默认返回

```js
onBackPress(options) {
    //获取上个页面的实例
    let pages = getCurrentPages(); 
    let prevPage = pages[pages.length - 2];
    //调用上一个页面的方法或者改变值
    prevPage.$vm.fun() 
    if (options.from === "navigateBack") {
      return false;
    }
    this.goback();
    return true;
  },
  methods: {   
      goback() {  
          //返回上一页
           uni.navigateBack({  
               delta: 1  
            });  
        }  
    },   
      
```
